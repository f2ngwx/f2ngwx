-size  大小---区间 。 大于 小于 等于 + 单位   find / -size +6M -10M 
-type  类型 f d l b 
-atime 根据时间单位，第一组默认天   a是记录最后一次访问时间
-ctime 记录最后一次属性上的时间
-mtime 记录最后一次内容上的修改
-amin  第二章默认分钟
-cmin
-mmin
-user  属主
-group 属组
-perm  权限
  • 在linux下找以a开头，以.log结尾，排除access.log，在10天以前修改，大于5M，小于20M，用户为root的普通文件。同时复制到/opt/下
find / -name "a*.log"  ! -name access.log -mtime -10 -size +5M -size -20M -user root -type f -exec cp -a {} /opt/ \;
或 | xargs -i cp{} /opt

SED
  • sed是流编辑器，处理时把当前处理的行存储在临时缓冲区，称为模式空间，接着用sed命令处理缓冲区中的内容，处理完成，把缓冲区的内容送往屏幕。接着处理下一行，不断重复直到文件末尾。文件内容没有改变，除非你用重定向存储输出。对文件进行增删查改
sed [options] 'command' file(s)
sed [options] -f scriptfile file(s)
  • sed命令通过格式方式表示，比如 sed -i 's///d' 中的d就叫做子命令
  • sed参数
-e 默认选项，在命令行模式进行sed动作编辑
-f  将sed动作写在一个文件内，用filename内的sed执行动作
-i  直接修改文件内容，跳过缓冲区
-n  只打印模式匹配的行
-r  支持扩展表达式
  • sed常用命令
a\    在当前行下面插入文本
i\    在当前行上面插入文本
c\    把选定的行改为新的文本

  • 删除行
sed 's/^$//d' file  删除空白行
sed '2d' file
sed '2,$d' file 删除文件第二行到末尾所有行
sed '/^test/'d file  删除所有开头是 test的行
  • 正则表达式 \w\+ 匹配每一个单词，使用 [&] 替换他，&对应于之前所匹配到的单词
echo this is a test line | sed 's/\w\+/[&]/g'
  • 子串匹配标记
echo this is digit7 in a number | sed 's/digit\([0-9]\)/\1/'
echo aaa BBB | sed 's/\([a-z]\+\) \([A-Z]\+\)/\2 \1/'
sed -n  's/\(love\)able/\1rs/p' file
  • 选定行的范围
sed -n '/test/,/check/p' file   # 所有在模板test和check所确定范围内的行都被打印
sed -n '5,/^test/p' file   #打印从第5行到第一个包含以test开始的行之间的所有行
sed -n '/test/,/west/s/$/aaa bbb/' file  #对于模板test和west的行，每行的末尾用字符串aaa bbb替换
  • 多点编辑 。-e选项允许在同一行里执行多条命令
sed -e '1,5d' -e 's/test/check/' file   # 前面会影响后面的执行结果
  • 从文件读入：r命令。file里的内容被读进来，显示在与test匹配的行后面，如果匹配多行，则file的内容将显示在所有匹配行的下面。
sed '/test/r file'  filename
  • 追加，a\命令
sed '/^test/a\this is a test line' file
sed -i '2a\this is a test line' test.conf
  • 插入（行上）
sed '/^test/i\this is a test line' file
  • n命令
sed '/test/{ n;s/aa/bb/; }'   file  #如果test被匹配，则移动到匹配行的下一行，替换这行的aa变为bb


AWK
  • awk是编程语言，支持变量、函数、流控语句（if，while，for）、数据类型。目前使用的版本是gawk。
  • awk用于匹配内容，与grep相同，都是以行作为单位。不同处是awk比grep更精准，可以将行分成端进行精确匹配。
awk [options] 'commands' file(s)
awk [options] -f scriptfile file(s)
  • 参数
-F   定义字段分割符号     $0 等于整行
-v   定义变量并赋值
  • BEGIN和END是已有的函数
BEGIN{}{正则|基本|if等等}   END{print...}
  • 命令格式
awk '/root/'  /etc/passwd
awk -F:  '{print $1}' /etc/passwd
awk 'BEGIN{FS=":"} /root/{print $1,$3}' /etc/passwd
df -P | grep '/' | awk '$4 > 25000 {print $4}'
  • 相关变量
$0    文本本身
NF    列数
$NF    最后一列
$(NF-1)    倒数第二列
FNR/NR    行号
FILENAME    文件名
"\t"    制表符
RS    换行符
""    打印字符串
FS    定义间隔符
~    匹配
!~    不匹配
==    完全相等
/[0-9][0-9]+/   两个或两个以上数字
-F'[:#/]'    定义三个分隔符
  • 参数说明
%s 字符类型
%d  数值类型
-   表示左对齐，默认是右对齐
printf    默认不会在行尾加换行\n
  • awk模式和动作：任何awk语句都由模式和动作组成。模式部分决定动作语句何时触发事件。处理即对数据进行的操作。如果省略部分，动作将时刻保持执行状态。模式可以是任何条件语句或符合语句或正则表达式。模式包括两个特殊字段BEGIN和END。使用BEGIN语句设置计数和打印头，。
    • 模式可以是：正则表达式、比较表达式、条件表达式、算术运算符、逻辑操作符、复合模式、范围模式
  • 正则
awk '/^root/' /etc/passwd
awk '!/root/'  /etc/passwd    
awk -F: '$1 ~ /^root/' /etc/passwd
awk -F: '$NF !~ /nologin$/'  /etc/passwd
  • 比较运算符
awk -F: '$3==0' /etc/passwd
awk -F: '$3<10' /etc/passwd
awk -F: '$7=="/bin/bash"' /etc/passwd
awk -F: '$1=="root"' /etc/passwd
awk -F: '$1 ~ /admin/' /etc/passwd
  • 条件表达式
awk -F: '$3>300 {print $0}' /etc/passwd
awk -F: '{ if($3>300) print $0}'  /etc/passwd
awk -F: '{ if($3>300){print $0}}'  /etc/passwd
awk -F: '{ if($3>300){print $3} else{print $1}}' /etc/passwd
  • 算术运算  + - *  / % ^
awk -F: '$3 * 10 >500' /etc/passwd
  • 逻辑操作符和复合模式  && ||  ！
awk -F: '$3 > 50 && $3 <= 100' /etc/passwd
awk -F: '$3 == 0 || $3 <= 10' /etc/passwd
awk -F: '!($3==0||$3<=10)' /etc/passwd
  • 条件判断
awk -F: '{if(NR>=10 && NR<=20) print $1}'  /etc/passwd
awk -F: '{if($3==0){print $1 " is administrator."}}' /etc/passwd
awk -F: '{if($3>0 && $3<1000){count++;}} END{print count}' /etc/passwd
BEGIN{sum=0}{++sum[$NF]}END{for(i in sum){print sum[i]"\t"i}}
  • while循环
awk 'BEGIN{i=1;while(i<=10){print i;i++}}'
awk -F: '{i=1;while(i<=10){print $0;i++}}' /etc/passwd
  • for循环
awk 'BEGIN{for(i=1;i<=5;i++){print i}}'
  • 数组
awk -F: '{username[++i]=$1} END{print username[1]}' /etc/passwd
awk -F: '{username[i++]=$1} END{print username[1]}'  /etc/passwd
  • 数组遍历
awk -F: '{username[x++]=$1} END{for(i=0;i<x;i++) print i,username[i]}'  /etc/passwd
